Event Listener және интерактивтілік

JavaScript Event Listener арқылы HTML элементтерімен интерактивті әрекеттер жасауға болады. Мысалы, батырманы басқанда, форманы өзгерткенде, тінтуірді жылжытып жүргенде т.б. әрекеттер орындалады.

1. Event Listener қосу

// Батырмаға click оқиғасын қосу
let button = document.getElementById("btn");

button.addEventListener("click", function() {
  alert("Батырма басылды!");
});

// Мышка үстінде hover
let box = document.getElementById("box");
box.addEventListener("mouseenter", function() {
  box.style.backgroundColor = "yellow";
});
box.addEventListener("mouseleave", function() {
  box.style.backgroundColor = "lightblue";
});

2. Input және Form оқиғалары

let input = document.getElementById("name");

input.addEventListener("input", function() {
  console.log("Input мәні:", input.value);
});

let form = document.getElementById("myForm");
form.addEventListener("submit", function(event) {
  event.preventDefault(); // форма жіберуді тоқтату
  alert("Форма жіберілді: " + input.value);
});
